from django.shortcuts import render
from .models import Post
from django.shortcuts import get_object_or_404, redirect, reverse
from .forms import *
from django.views import generic


def post_view_of_blog(request):
    all_posts = Post.objects.filter(status='pub').order_by('-datetime_modified')
    post_list_dic = {
        'post_list': all_posts
    }
    return render(request, 'blog/post_list.html', post_list_dic)


def post_detail_view(request, pk):
    post = get_object_or_404(Post, pk=pk)
    return render(request, 'blog/post_detail.html', {'post': post})


def showing_add_post_page(request):
    if request.method == 'POST':
        form = NewPostFrom(request.POST)
        if form.is_valid():
            form.save()
            return redirect(reverse('post_view_of_blog'))
    else:
        form = NewPostFrom()
    return render(request, 'blog/add_post.html', context={'form_1': form})


def edit_posts(request, pk):
    post = get_object_or_404(Post, pk=pk)
    form = NewPostFrom(request.POST or None, instance=post)
    if form.is_valid():
        form.save()
        return redirect(reverse('post_view_of_blog'))
    return render(request, 'blog/add_post.html', context={'form_1': form})


def delete_post(request, pk):
    post = get_object_or_404(Post, pk=pk)
    if request.method == 'POST':
        post.delete()
        return redirect(reverse('post_view_of_blog'))
    if request.method == 'GET':
        return render(request, 'blog/delete_post.html', context={'post': post})
